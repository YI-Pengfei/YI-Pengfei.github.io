BFS


BFS基本代码：

import networkx as nx

G = nx.Graph()
G.add_edge(1,2)
G.add_edge(2,4)
G.add_edge(2,3)
G.add_edge(3,5)

def bfs(G,node):
    res = []
    visited = {}
    for node in G.nodes(): # 标记所有节点为 "未访问"
        visited[node] = False
    queue = [node]
    visited[node] = True
    while queue:
        cur_node = queue.pop(0) # 队头出列
        res.append(cur_node)
        for child in nx.neighbors(G, cur_node):
            if not visited[child]:
                visited[child]=True
                queue.append(child) # 入队尾
    return res

print(bfs(G,5))
